CREATE DATABASE IF NOT EXISTS boost;
USE boost;

-- DADOS CADASTRAIS

CREATE TABLE aluno (
    id INT AUTO_INCREMENT PRIMARY KEY,
    nome VARCHAR(100),
    email VARCHAR(200),
    senha VARCHAR(200),
    telefone CHAR(11),
    tipo VARCHAR(100)
);

CREATE TABLE professor (
    id INT AUTO_INCREMENT PRIMARY KEY,
    nome VARCHAR(100),
    email VARCHAR(200),
    senha VARCHAR(200),
    telefone CHAR(11),
    tipo VARCHAR(100)
);

CREATE TABLE turma (
    id INT AUTO_INCREMENT PRIMARY KEY,
    nome VARCHAR(45),
    nivel VARCHAR(50),
    limite_alunos INT
);

-- MANY TO MANY: aluno ↔ turma

CREATE TABLE aluno_turma (
    aluno_id INT NOT NULL,
    turma_id INT NOT NULL,

    PRIMARY KEY (aluno_id, turma_id),

    CONSTRAINT fk_aluno
        FOREIGN KEY (aluno_id) REFERENCES aluno(id),

    CONSTRAINT fk_turma
        FOREIGN KEY (turma_id) REFERENCES turma(id)
);

select * from turma;
select * from aluno;
select * from professor;
select * from aluno_turma;


drop database boost;
